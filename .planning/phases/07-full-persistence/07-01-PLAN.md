```javascript
// Saving:
function saveConnections() {
  const serializable = connections.map(c => ({
    fromIdx: c.from.dataset.index,
    toIdx: c.to.dataset.index
  }));
  localStorage.setItem('investigation_connections', JSON.stringify(serializable));
}

// Restoring:
const savedConns = JSON.parse(localStorage.getItem('investigation_connections') || '[]');
savedConns.forEach(data => {
  const from = document.querySelector(`.card[data-index="${data.fromIdx}"]`);
  const to = document.querySelector(`.card[data-index="${data.toIdx}"]`);
  createConnection(from, to);
});
```
