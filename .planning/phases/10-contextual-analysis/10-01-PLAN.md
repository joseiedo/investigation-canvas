**Highlighting helper:**
```javascript
function highlightSelection() {
  const selection = window.getSelection();
  if (!selection.rangeCount) return;
  const range = selection.getRangeAt(0);
  const span = document.createElement('span');
  span.className = 'highlight';
  range.surroundContents(span);
  
  cardHighlights[currentCardIndex] = dialogTxt.innerHTML;
  const boardCard = document.querySelector(`.card[data-index="${currentCardIndex}"]`);
  boardCard.querySelector('.card-text').innerHTML = dialogTxt.innerHTML;
  save();
}
```
