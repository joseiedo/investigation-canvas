<task type="auto">
  <name>Task 2: Add inline JavaScript for card dragging</name>
  <files>index.html</files>
  <action>
Add a `<script>` block before the closing `</body>` tag in `index.html` to handle the drag-and-drop logic.

**Implementation requirements:**
1. Use `mousedown` on individual `.card` elements to initiate dragging.
2. Use `window.addEventListener('mousemove', ...)` and `window.addEventListener('mouseup', ...)` while dragging to ensure the card follows the cursor even if it leaves the card's bounds.
3. Calculate the offset (mouse click vs. card's top-left) to prevent the card from "snapping".
4. Update the card's `style.left` and `style.top` in pixels during `mousemove`.
5. Ensure the `z-index` remains high after the drag so the most recently moved card is on top of older ones.

**JavaScript structure:**
```javascript
<script>
  let activeCard = null;
  let offset = { x: 0, y: 0 };
  let highestZ = 100;

  document.querySelectorAll('.card').forEach(card => {
    card.addEventListener('mousedown', (e) => {
      activeCard = card;
      
      // Bring to front
      highestZ++;
      card.style.zIndex = highestZ;
      card.classList.add('dragging');
      
      // Calculate mouse offset from card top-left
      const rect = card.getBoundingClientRect();
      offset.x = e.clientX - rect.left;
      offset.y = e.clientY - rect.top;
      
      // Add global listeners
      window.addEventListener('mousemove', onMouseMove);
      window.addEventListener('mouseup', onMouseUp);
    });
  });

  function onMouseMove(e) {
    if (!activeCard) return;
    
    // Prevent default to avoid scroll/selection issues
    e.preventDefault();
    
    // Calculate new position relative to #board
    const board = document.getElementById('board');
    const boardRect = board.getBoundingClientRect();
    
    let x = e.clientX - boardRect.left - offset.x;
    let y = e.clientY - boardRect.top - offset.y;
    
    // Boundary constraints (optional, but prevents cards leaving board)
    const maxX = boardRect.width - activeCard.offsetWidth;
    const maxY = boardRect.height - activeCard.offsetHeight;
    
    x = Math.max(0, Math.min(x, maxX));
    y = Math.max(0, Math.min(y, maxY));
    
    activeCard.style.left = x + 'px';
    activeCard.style.top = y + 'px';
  }

  function onMouseUp() {
    if (activeCard) {
      activeCard.classList.remove('dragging');
      activeCard = null;
    }
    
    window.removeEventListener('mousemove', onMouseMove);
    window.removeEventListener('mouseup', onMouseUp);
  }
</script>
```
