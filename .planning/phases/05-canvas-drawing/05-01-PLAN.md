```javascript
function mirrorToBoard() {
  if (currentCardIndex === null) return;
  const cardCanvas = document.querySelectorAll('.card')[currentCardIndex].querySelector('.card-canvas');
  const ctx = cardCanvas.getContext('2d');
  ctx.clearRect(0, 0, cardCanvas.width, cardCanvas.height);
  ctx.drawImage(expandedCanvas, 0, 0, cardCanvas.width, cardCanvas.height);
}
```
